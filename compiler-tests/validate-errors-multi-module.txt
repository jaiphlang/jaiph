=== duplicate import alias
# @expect error E_VALIDATE "duplicate import alias"
--- main.jh
import "a.jh" as mod
import "b.jh" as mod

workflow default {
  ensure mod.one()
}
--- a.jh
script one_impl {
  echo one
}
rule one {
  run one_impl()
}
--- b.jh
script two_impl {
  echo two
}
rule two {
  run two_impl()
}
