=== duplicate import alias
# @expect error E_VALIDATE "duplicate import alias" @2:1
--- main.jh
import "a.jh" as mod
import "b.jh" as mod

workflow default() {
  run mod.one()
}
--- a.jh
script one_impl = `echo one`
workflow one() {
  run one_impl()
}
--- b.jh
script two_impl = `echo two`
workflow two() {
  run two_impl()
}
